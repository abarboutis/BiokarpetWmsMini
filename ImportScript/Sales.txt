	[IMPORT]
MODULE=TSTORECUSTTRADE
DESTINATION=FINTRADE,STORETRADE,STORETRADELINES
READER=ATLANTIS

[ATLANTIS]
MAINTABLE=QRY1

[IMPORTSET]
QRY1=SELECT DISTINCT COMID,BRANCHID,ID,FTRDATE, DSRID, ERPCustomerID,TransCode, 
STOREID,1 AS LOCALRATE,10 AS CURID FROM ZIMPORTHEADER2 WHERE  STATUS= 1 ORDER BY  ID;
QRY2=SELECT  ITEMID,LOTID,ZLENGTH,ZWIDTH,ZCOLOR,ZDRAFT,QTY,QTY2 FROM ZIMPORTDETAIL2
WHERE  STATUS = 1 AND IMPID=:$ID;

[FINTRADE]
DSRID=QRY1.DSRID
CUSID=QRY1.ERPCUSTOMERID
FTRDATE=QRY1.FTRDATE
TRADECODE=QRY1.TRANSCODE
CURID=QRY1.CURID
LOCALRATE=QRY1.LOCALRATE

[STORETRADE]
STOID=QRY1.STOREID

[STORETRADELINES]
ITEID=QRY2.ITEMID
PARID=QRY2.LOTID
PRIMARYQTY=QRY2.QTY2
SECONDARYQTY=QRY2.QTY
ZWIDTH=QRY2.ZWIDTH
ZLENGTH=QRY2.ZLENGTH
ZDRAFT=QRY2.ZDRAFT
ZCOLOR=QRY2.ZCOLOR


[AFTERIMPORT]
UPDATE ZIMPORTHEADER2 SET STATUS=0
UPDATE ZIMPORTDETAIL2 SET STATUS=0
