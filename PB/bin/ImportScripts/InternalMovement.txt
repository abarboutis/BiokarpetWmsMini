[IMPORT]
MODULE=TInternalTransfer
DESTINATION=FINTRADE,STORETRADE,STORETRADELINES
READER=ATLANTIS

[ATLANTIS]
MAINTABLE=QRY1

[IMPORTSET]
QRY1=SELECT DISTINCT ERPCompID, AlpisStoreTransID,AlpisStoreTransDate, ERPTransSeriesID, ERPSupplierID, ERPTransCode, 
ERPFromStoreID,ERPToStoreID,1 AS LOCALRATE,10 AS CURID,5 AS BRAID,NULL AS BINCODE,NULL AS BINSECCODE,0 AS  APPROVED,
ERPProdOrderID  FROM ZALPISTORESTRANS 
WHERE  isnew = 1 ORDER BY  AlpisStoreTransID;
QRY2=SELECT  ERPItemID, ERPItemMUnit, ERPItemQty, ERPItemUnitPrice, ERPItemQtyPrice,ERPLotID,
ERPFromStoreID,ERPToStoreID FROM ZALPISTORESTRANS 
WHERE  isnew = 1 AND AlpisStoreTransID=:$AlpisStoreTransID;

[FINTRADE]
DSRID=QRY1.ERPTransSeriesID
TRADECODE=QRY1.ERPTransCode
FTRDATE=QRY1.AlpisStoreTransDate
[STORETRADE]
STOID=QRY1.ERPFromStoreID
SECSTOID=QRY1.ERPToStoreID
PRODORDID=QRY1.ERPProdOrderID
[STORETRADELINES]
ITEID=QRY2.ERPItemID
PARID=QRY2.ERPLotID
PRIMARYQTY=QRY2.ERPItemQty
STOID=QRY2.ERPFromStoreID
SECSTOID=QRY2.ERPToStoreID
[AFTERIMPORT]
UPDATE ZALPISTORESTRANS SET ISNEW=0;
